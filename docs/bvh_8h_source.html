25 template <typename UInt>
26 void construct_internal_nodes(node* nodes, UInt const* node_code, const uint32_t num_objects)
27 {
28 thrust::for_each(
29 thrust::device,
30 thrust::make_counting_iterator<uint32_t>(0),
31 thrust::make_counting_iterator<uint32_t>(num_objects - 1),
32 [nodes, node_code, num_objects] __device__ (const uint32_t idx)
33 {
34 nodes[idx].object_idx = 0xFFFFFFFF; // internal nodes
35
36 const uint2 ij = determine_range(node_code, num_objects, idx);
37 const int gamma = find_split(node_code, num_objects, ij.x, ij.y);
38
39 nodes[idx].left_idx = gamma;
40 nodes[idx].right_idx = gamma + 1;
41 if(thrust::min(ij.x, ij.y) == gamma)
42 {
43 nodes[idx].left_idx += num_objects - 1;
44 }
45 if(thrust::max(ij.x, ij.y) == gamma + 1)
46 {
47 nodes[idx].right_idx += num_objects - 1;
48 }
49 nodes[nodes[idx].left_idx].parent_idx = idx;
50 nodes[nodes[idx].right_idx].parent_idx = idx;
51 return;
52 });
53 }

56template <typename Real, typename Object>
57struct default_morton_code_calculator
58{
59 default_morton_code_calculator(AABB<Real> w)
60 : whole(w)
61 {
62 }
63 default_morton_code_calculator() = default;
64 ~default_morton_code_calculator() = default;
65 default_morton_code_calculator(default_morton_code_calculator const&) = default;
66 default_morton_code_calculator(default_morton_code_calculator&&) = default;
67 default_morton_code_calculator& operator=(default_morton_code_calculator const&) = default;
68 default_morton_code_calculator& operator=(default_morton_code_calculator&&) = default;
69
70 __device__ __host__ inline uint32_t operator()(const Object&, const AABB<Real>& box) noexcept
71 {
72 auto p = centroid(box);
73 p.x -= whole.lower.x;
74 p.y -= whole.lower.y;
75 p.z -= whole.lower.z;
76 p.x /= (whole.upper.x - whole.lower.x);
77 p.y /= (whole.upper.y - whole.lower.y);
78 p.z /= (whole.upper.z - whole.lower.z);
79 return morton_code(p);
80 }
81 AABB<Real> whole;
82};

253 // --------------------------------------------------------------------
254 // create AABB for each node by bottom-up strategy
255
256 m_flag_container.clear();
257 m_flag_container.resize(num_internal_nodes, 0);
258 const auto flags = m_flag_container.data();
259
260 thrust::for_each(
261 thrust::device,
262 thrust::make_counting_iterator<index_type>(num_internal_nodes),
263 thrust::make_counting_iterator<index_type>(num_nodes),
264 [nodes = m_nodes.data(), aabbs = m_aabbs.data(), flags] __device__(index_type idx)
265 {
266 uint32_t parent = nodes[idx].parent_idx;
267 while(parent != 0xFFFFFFFF) // means idx == 0
268 {
269 const int old = atomicCAS(flags + parent, 0, 1);
270 if(old == 0)
271 {
272 // this is the first thread entered here.
273 // wait the other thread from the other child node.
274 return;
275 }
276 MUDA_KERNEL_ASSERT(old == 1);
277 // here, the flag has already been 1. it means that this
278 // thread is the 2nd thread. merge AABB of both childlen.
279
280 const auto lidx = nodes[parent].left_idx;
281 const auto ridx = nodes[parent].right_idx;
282 const auto lbox = aabbs[lidx];
283 const auto rbox = aabbs[ridx];
284 aabbs[parent] = merge(lbox, rbox);
285
286 // look the next parent...
287 parent = nodes[parent].parent_idx;
288 }
289 return;
290 });
291 }
