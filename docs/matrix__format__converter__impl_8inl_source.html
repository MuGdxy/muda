21template <typename T>
22void MatrixFormatConverter<T, 1>::merge_sort_indices_and_values(
23 const DeviceTripletMatrix<T, 1>& from, DeviceBCOOMatrix<T, 1>& to)
24{
25 using namespace muda;
26
27 auto src_row_indices = from.row_indices();
28 auto src_col_indices = from.col_indices();
29 auto src_values = from.values();
30
31 sort_index.resize(src_row_indices.size());
32 ij_pairs.resize(src_row_indices.size());
33
34 ParallelFor(256)
35 .kernel_name("set ij pairs")
36 .apply(src_row_indices.size(),
37 [row_indices = src_row_indices.cviewer().name("row_indices"),
38 col_indices = src_col_indices.cviewer().name("col_indices"),
39 ij_pairs = ij_pairs.viewer().name("ij_pairs")] __device__(int i) mutable
40 {
41 ij_pairs(i).x = row_indices(i);
42 ij_pairs(i).y = col_indices(i);
43 });
44
45 ParallelFor(256)
46 .kernel_name("iota") //
47 .apply(src_row_indices.size(),
48 [sort_index = sort_index.viewer().name("sort_index")] __device__(int i) mutable
49 { sort_index(i) = i; });
50
51 DeviceMergeSort().SortPairs(workspace,
52 ij_pairs.data(),
53 sort_index.data(),
54 ij_pairs.size(),
55 [] __device__(const int2& a, const int2& b) {
56 return a.x < b.x || (a.x == b.x && a.y < b.y);
57 });
58
59 // set ij_pairs back to row_indices and col_indices
60
61 auto dst_row_indices = to.row_indices();
62 auto dst_col_indices = to.col_indices();
63
64 ParallelFor(256)
65 .kernel_name("set col row indices")
66 .apply(dst_row_indices.size(),
67 [row_indices = dst_row_indices.viewer().name("row_indices"),
68 col_indices = dst_col_indices.viewer().name("col_indices"),
69 ij_pairs = ij_pairs.viewer().name("ij_pairs")] __device__(int i) mutable
70 {
71 row_indices(i) = ij_pairs(i).x;
72 col_indices(i) = ij_pairs(i).y;
73 });
74
75 // sort the block values
76
77 unique_values.resize(from.m_values.size());
78
79 ParallelFor(256)
80 .kernel_name("set block values")
81 .apply(src_values.size(),
82 [src_values = src_values.cviewer().name("blocks"),
83 sort_index = sort_index.cviewer().name("sort_index"),
84 dst_values = unique_values.viewer().name("values")] __device__(int i) mutable
85 { dst_values(i) = src_values(sort_index(i)); });
86}

135 DeviceCOOMatrix<T>& to)
136{
137 using namespace muda;
138
139 auto& row_indices = to.m_row_indices;
140 auto& values = to.m_values;
141 values.resize(row_indices.size());
142 // first we add the offsets to counts, to get the offset_ends
143

146 .apply(unique_counts.size(),

207void MatrixFormatConverter<T, 1>::calculate_block_offsets(const DeviceCOOMatrix<T>& from,
208 DeviceCSRMatrix<T>& to)
209{
210 using namespace muda;
211 to.reshape(from.rows(), from.cols());
212
213 auto& dst_row_offsets = to.m_row_offsets;
214
215 // alias the offsets to the col_counts_per_row(reuse)
216 auto& col_counts_per_row = offsets;
217 col_counts_per_row.resize(to.m_row_offsets.size());
218 col_counts_per_row.fill(0);
219
220 unique_indices.resize(from.non_zeros());
221 unique_counts.resize(from.non_zeros());
222
223 // run length encode the row
224 DeviceRunLengthEncode().Encode(workspace,
225 from.m_row_indices.data(),
226 unique_indices.data(),
227 unique_counts.data(),
228 count.data(),
229 from.non_zeros());
230 int h_count = count;
231
232 unique_indices.resize(h_count);
233 unique_counts.resize(h_count);
234
235 ParallelFor(256)
236 .kernel_name(__FUNCTION__)
237 .apply(unique_counts.size(),
238 [unique_indices = unique_indices.cviewer().name("offset"),
239 counts = unique_counts.viewer().name("counts"),
240 col_counts_per_row = col_counts_per_row.viewer().name(
241 "col_counts_per_row")] __device__(int i) mutable
242 {
243 auto row = unique_indices(i);
244 col_counts_per_row(row) = counts(i);
245 });
246
